	var book={
		name: "OOP in JS", 
		_price: 0, 
		get price(){return this._price + "$";},
		set price(val){
			if (val < 0)
				window.alert("Invalid price");
			this._price = val;}
	};	
	book.price = 1000;// run set price(val)...
	window.alert("name is "+ book.name +", price is "+ book.price);//get price()...
	book.price = -100;// run set price(val)...
	book._price = -100; 
	window.alert("name is "+ book.name +", price is "+ book.price);//get price()...
