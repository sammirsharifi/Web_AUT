CSS & JavaScript Example

 

var color;
var index = prompt("Enter 1 for blue, 2 for red, 3 for green");
switch(parseInt(index)) {
  case 1:
    color = "blue"; break;
  case 2:
    color = "red"; break;
  case 3:
    color = "green"; break;
  default:
    color = "black"; break;
}

document.body.style.backgroundColor = color;
/*or
(document.getElementsByTagName("body"))[0].style.backgroundColor = color;
*/
