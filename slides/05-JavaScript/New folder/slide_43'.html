JavaScript Form Validation Exmaple

 

<form>
<textarea rows="3" cols="20" id="txtID"> </textarea>
<input type="button" value="Convert" onclick="upperCaseArea('txtID', 'outID')" />
</form>
Output will be here: <span id="outID" style="border-style: solid; width:50%"> &nbsp; </span>
Output will be here:  
<script type="text/javascript">
  function upperCaseArea(textareaID, outputID) {
    var textareaObject = document.getElementById(textareaID);
    var content = textareaObject.value.toUpperCase();
    var outputObject = document.getElementById(outputID);
    var outputMessage = content;
    outputObject.innerHTML = outputMessage;
  }
</script>
